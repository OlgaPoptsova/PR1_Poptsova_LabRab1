            var a,b,c,d,k,l,m,n;
            a=prompt('Введите число a', 0);
            b=prompt('Введите число b', 0);
            c=prompt('Введите число c', 0);
            d=prompt('Введите число d', 0);
             +a;
             +b;
             +c;
             +d;
             +k;
             +l;
             +m;
             +n;
            if (a<=b & b<=c & c<=d) {
               k=d, l=d, m=d, n=d
               } else if (a>b & b>c & c>d) {
               k=a, l=b, m=c, n=d
               } else {
               k=a*a, l=b*b, m=c*c, n=d*d
               }
            alert(k+' '+l+' '+m+' '+n);
         </script>
